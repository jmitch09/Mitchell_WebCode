<head>
    <title>Problem 6</title>
    <script>
        var apples= {
            item: "apples",
            price: 1,
            quantity: 0
        }

        var oranges= {
            item: "oranges",
            price: 2,
            quantity: 0
        }

        var peanuts= {
            item: "peanuts",
            price: 3,
            quantity: 0
        }

        var grapes= {
            item: "grapes",
            price: 2,
            quantity: 0
        }

        var popcorn= {
            item: "popcorn",
            price: 3.00,
            quantity: 0
        }

        food = [apples, oranges, peanuts, grapes, popcorn]
        
        total = 0

        function startOver() {
            food.forEach(element => {
                element.quantity = 0
            });
            total = 0

            document.getElementById("result").innerHTML =
            "Total is $" + total
        }

        function compute() {

            index = document.getElementById("items").selectedIndex
            if (index >= 0) {
                item = food[index]
            } else {
                alert("Select An Item")
            }

            quan = document.getElementById("quantity").value
            parseInt(quan)
            if ((quan < 1) || (quan > 100)) {
                alert("Invalid Quantity")
            }

            parseInt(item.quantity)
            item.quantity += quan

            cost = parseInt(item.price) * quan

            total += cost

            document.getElementById("result").innerHTML =
               item.item + " " + quan + " $" + cost + "<br />" + "Total is $" + total
           
        }
    </script>
</head>
<body>
    <h1>Problem 6</h1>
   
    <form>
    <label>Options:</label>
        <select id="items" name="Items">
        </select> <br />
    <script>
        list = document.getElementById("items")
        i = 0
        while (i < 5) {
            option = document.createElement("option")
            option.text = food[i].item
            option.value = food[i].item
            list.appendChild(option)
            i++
        }
    </script>

    <label>Quantity:</label><input type="text" name="quantity" id="quantity"> <br />
    <input type="button" value="Add" onclick="compute()">
    <input type="button" value="Start Over" onclick="startOver()">
</form>
   
<div id="result">&nbsp;</div>
</body>
