<head>
    <title>Problem 5</title>
    <script>
        total = 0
        items = ["apples", "oranges", "peanuts", "grapes", "popcorn"]
        prices = ["1", "2", "3", "2", "3"]

        function startOver() {
            total = 0
            document.getElementById("result").innerHTML =
            "Total is $" + total
        }

        function compute() {

            index = document.getElementById("items").selectedIndex
            if (index >= 0) {
                item = items[index]
            } else {
                alert("Select An Item")
            }

            quan = document.getElementById("quantity").value
            parseInt(quan)
            if ((quan < 1) || (quan > 100)) {
                alert("Invalid Quantity")
            }

            cost = parseInt(prices[index]) * quan

            total += cost

            document.getElementById("result").innerHTML =
               item + " " + quan + " $" + cost + "<br />" + "Total is $" + total
            
        }
    </script>
</head>
<body>
    <h1>Problem 5</h1>
  
    <form>
        <label>Options:</label>
            <select id="items" name="Items">
            </select> <br />
        <script>
            list = document.getElementById("items")
            i = 0
            while (i < 5) {
                option = document.createElement("option")
                option.text = items[i]
                option.value = items[i]
                list.appendChild(option)
                i++
            }
        </script>

        <label>Quantity:</label><input type="text" name="quantity" id="quantity"> <br />
        <input type="button" value="Add" onclick="compute()">
        <input type="button" value="Start Over" onclick="startOver()">
    </form>
       
    <div id="result">&nbsp;</div>
    
</body>
